  def allHeaders
    mailHeader = ""
    # --add CC
    mailHeader += "CC: " + @cc + "\n" if (@cc != "")
    # --add BCC
    mailHeader += "BCC: " + @bcc + "\n" if (@bcc != "")
    # --add From
    mailHeader += "From: " + @from + "\n" if (@from != "")
    # --add extras
    @headers.each { |key, val|
      mailHeader += key + ": " + val + "\n"
    }
    mailHeader
  end
