  def parseFrom(fromIn)
    from = name = nil
    return from, name if fromIn.nil? || fromIn.empty?
    ary = fromIn.split
    from = ary.pop
    if from && from[0] == ?<
      from = from[1..-2]
    end
    name = ary.join(' ') if ary.size > 0
    name = name[1..-2] if name && name[0] == ?"
    return from, name
  end
